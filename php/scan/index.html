function redis_init($db=0){
    $host = getenv("REDIS_HOST");
    $port = getenv("REDIS_PORT");
    
    $redis = new Redis();
    $redis->connect($host,$port);
    $redis->select($db);
    
    return $redis;
}

	$match = sprintf("test:%s:num", $cdate);
	$key_list = $redis->scan($cur, $match, 5);
	# $key_list 有可能是空数组，空数组和false是相等的。
	# 如果结果为false，需要用===比较 （类型和值都相等）。
	if($key_list === false){
		break;
	}
	foreach($key_list as $key){
		echo $key."\n";
	}
	
}while($cur);

// 也可以while条件是true，在代码中判断$cur 游标==0时退出。

$redis->close();
